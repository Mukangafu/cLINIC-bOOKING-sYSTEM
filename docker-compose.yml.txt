version: '3.8'
services:
  db:
    image: mysql:8.0
    container_name: clinic_db
    environment:
      MYSQL_ROOT_PASSWORD: rootpass
      MYSQL_DATABASE: clinic_db
    ports:
      - "3307:3306"
    volumes:
      - db_data:/var/lib/mysql

  api:
    build: .
    container_name: clinic_api
    environment:
      DB_HOST: db
      DB_PORT: 3306
      DB_USER: root
      DB_PASSWORD: rootpass
      DB_NAME: clinic_db
      PORT: 3000
    ports:
      - "3000:3000"
    depends_on:
      - db
    command: sh -c "npm install && npm run dev"

volumes:
  db_data:
